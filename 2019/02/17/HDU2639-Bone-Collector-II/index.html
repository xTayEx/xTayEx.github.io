#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#define mst(a,b) memset(a,b,sizeof(a))
using namespace std;
typedef long long ll;
int val[1005];
int wei[1005];
int a[1005],b[1005]; 
int dp[1005][31];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {    
        mst(dp,0);
        int n,vtot,k;
        scanf("%d %d %d",&n,&vtot,&k);
        for(int i=0;i<n;i++)        cin>>val[i];
        for(int i=0;i<n;i++)        cin>>wei[i];
        for(int i=0;i<n;i++)
        {
            for(int j=vtot;j>=wei[i];j--)
            {
                for(int q=0;q<k;q++)
                {
                    a[q]=dp[j][q];
                    b[q]=dp[j-wei[i]][q]+val[i];
                }
                a[k]=-1;b[k]=-1;
                int idx1=0;int idx2=0;int idx3=0;
                while((idx1<k||idx2<k)&&idx3<k)
                {
                    a[idx1]>b[idx2]?dp[j][idx3]=a[idx1++]:dp[j][idx3]=b[idx2++];
                    if(idx3==0||dp[j][idx3-1]!=dp[j][idx3])        //è¿™ä¸ªåˆ¤æ–­æ˜¯é˜²æ­¢æœ‰é‡å¤æ•°æ®
                        idx3++; 
                }
            }
        }
        cout<<dp[vtot][k-1]<<endl;
    }
    return 0;
}
