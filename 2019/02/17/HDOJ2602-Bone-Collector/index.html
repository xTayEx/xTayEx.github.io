#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn=1005; 
int value[maxn],w[maxn];
int dp[maxn];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,vtot;
        scanf("%d %d",&n,&vtot);
        memset(value,0,sizeof(value));memset(w,0,sizeof(w));
        for(int i=1;i<=n;i++)        scanf("%d",&value[i]);
        for(int i=1;i<=n;i++)        scanf("%d",&w[i]);
        memset(dp,0,sizeof(dp));
        for(int i=1;i<=n;i++)
            for(int j=vtot;j>=w[i];j--)
                dp[j]=max(dp[j],dp[j-w[i]]+value[i]);
        int ans=dp[vtot];
        printf("%d\n",ans);
    }
    return 0;
}
