#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
typedef long long ll;
const int maxn=200000+5;
ll R[maxn];
ll sub[maxn];
ll dp[maxn];
int main()
{
    int n;
    memset(R,0,sizeof(R));
    memset(sub,0,sizeof(sub));
    memset(dp,0,sizeof(dp));
    scanf("%d",&n);
    for(int i=1;i<=n;i++)        scanf("%lld",&R[i]);
    for(int i=1;i<=n-1;i++)
        sub[i]=R[i+1]-R[i];
    dp[1]=sub[1];
    ll ans=INT_MIN;
    for(int i=2;i<=n-1;i++)
        dp[i]=max(dp[i-1]+sub[i],sub[i]);
    for(int i=1;i<=n-1;i++)
        if(dp[i]>ans)
            ans=dp[i];
    printf("%lld\n",ans);
    return 0;    
}
