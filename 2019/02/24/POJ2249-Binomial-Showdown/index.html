#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <algorithm>
#include <cmath>
#define INF 0x7fffffff
#define mst(a,b) memset(a,b,sizeof(a))
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
const int maxn=1e6+5;
ll ans[maxn];
int main()
{
    ll n,m;
    mst(ans,0);
    while(cin>>n>>m&&n){
        ans[0]=1;
        if(n-m<m)
            m=n-m;
        else{
            for(int i=1;i<=m;i++)
                ans[i]=(n-i+1)*ans[i-1]/i;
        }
        cout<<ans[m]<<endl;
    }
    return 0;
}
