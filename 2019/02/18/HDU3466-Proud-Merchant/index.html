#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef struct{
    int p,q,v;
}mer;
mer arr[505];
int dp[100000];
bool cmp(mer a,mer b)
{
    return (a.q-a.p)<(b.q-b.p);
}
int main()
{
    int n,m;
    while(cin>>n>>m)
    {    
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++)
            cin>>arr[i].p>>arr[i].q>>arr[i].v;
        sort(arr,arr+n,cmp);
        for(int i=0;i<n;i++)
            for(int j=m;j>=arr[i].q;j--)
                dp[j]=max(dp[j],dp[j-arr[i].p]+arr[i].v);
        cout<<dp[m]<<endl;
    }
    return 0;
}
