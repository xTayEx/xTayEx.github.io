int topo()
{
    queue<int> que;
    int cnt=0;
    for(int i=1;i<=n;i++){
        if(deg[i]==0){
            que.push(i);
            deg[i]--;

    int ans=0;
    while(!que.empty())
    {

        int fro=que.front();

            }
        }
    }

    return -1;
}
