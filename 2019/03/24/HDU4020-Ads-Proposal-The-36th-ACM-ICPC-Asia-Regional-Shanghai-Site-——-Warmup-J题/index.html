const int maxn=1e5+5;
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
struct ad{
    ll cli;
    ll len;
    bool operator<(const ad& a) const{
        return cli>a.cli;
    }
};
ll sum[maxn*5];
set<ad> cus[maxn];
int main()
{
    int t;
    scanf("%d",&t);
    int kase=0;
    while(t--)
    {
        int n,m,q;
        scanf("%d %d %d",&n,&m,&q);
        for(int i=0;i<=n;i++)
            cus[i].clear();
        for(int i=0;i<m;i++){
            ad tmp;int u;
            scanf("%d %lld %lld",&u,&tmp.cli,&tmp.len);
            cus[u].insert(tmp);
        }
        mst(sum,0);
        set<ad>::iterator it;
        for(int i=1;i<=n;i++){
            int idx=1;
            for(it=cus[i].begin();it!=cus[i].end();it++){
                sum[idx++]+=(*it).len;
            }
        }
        for(int i=1;i<=m;i++)
            sum[i]=sum[i]+sum[i-1];
        printf("Case #%d:\n",++kase);
        for(int i=0;i<q;i++){
            int k;
            scanf("%d",&k);
            k=min(k,m); 
            printf("%lld\n",sum[k]);
        }
    }
    return 0;
}
