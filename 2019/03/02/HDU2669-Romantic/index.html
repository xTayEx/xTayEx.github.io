ll exgcd(ll a,ll b,ll& x,ll& y)
{    
    ll r;
    if(b==0){    
        x=1;y=0;
        return a;
    }else{
        r=exgcd(b,a%b,y,x);
        y-=x*(a/b);
    }
    return r; 
}
ll solve(ll a,ll b)
{    
    ll x,y;
    ll g=exgcd(a,b,x,y);
    if(1%g)
        return -1;
    x*=1/g;
    if(b<0)        b=-b;
    ll ans=x%b;
    if(ans<=0)
        ans+=b;
    return ans;
}
int main()
{
    ll a,b;

    {
        ll ans=solve(a,b);
        if(ans==-1)
            cout<<"sorry"<<endl;

        {
            ll ansx=ans;
            ll ansy=(1-a*ansx)/b;
            cout<<ansx<<" "<<ansy<<endl;
        }
    }
    return 0;
}
