const int maxn=1e4+5;
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
int n1,n2,m;
int pre[maxn];
int cnt[maxn];
struct kid{
    int rcnt;
    int ccnt;
    kid():rcnt(0),ccnt(1){}
};
bool cmp(kid node1,kid node2)
{
    if(node1.ccnt==node2.ccnt)
        return node1.rcnt<node2.rcnt;
    return node1.ccnt<node2.ccnt;
}
void init(int n)
{
    for(int i=1;i<=n;i++)
        pre[i]=i;
}
int findr(int x)
{
    if(x==pre[x])        return pre[x];
    return pre[x]=findr(pre[x]);
}
int main()
{
    int t;
    cin>>t;
    int kase=0;
    while(t--)
    {    
        kid k1[maxn];
        kid k2[maxn];
        cin>>n1>>m;init(n1);
        for(int i=0;i<m;i++){
            int u,v;
            scanf("%d %d",&u,&v);
            int fu=findr(u);

            int fv=findr(v);
            if(fu!=fv){
                pre[fu]=fv;
                k2[fv].ccnt+=k2[fu].ccnt;
                k2[fu].ccnt=0;

                k2[fv].rcnt++;
