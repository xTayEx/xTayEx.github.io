#include <iostream>

#define debug printf("debug\n")
#define INF 0x3f3f3f3f
const int maxn=1e5+5;
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
int n,m; 
int pre[maxn];
struct edge{
    int from;
    int to;
    ll wei;
};
vector<int> tree[maxn];
edge arr[maxn*10];
ll sum[maxn];
ll ans; 
bool cmp(edge a,edge b)
{
    return a.wei<b.wei;
}
void init()
{
    for(int i=1;i<=n;i++)
        pre[i]=i;
}
int findr(int x)
{
    if(pre[x]==x)
        return pre[x];
    return pre[x]=findr(pre[x]);
}
ll kruskal()
{
    ll rsl=0;
    int num=0;
    init();
    for(int i=1;i<=m;i++)
    {
        int fx=findr(arr[i].from);
        int fy=findr(arr[i].to);
        if(fx!=fy)
        {
            pre[fx]=fy;
            rsl+=arr[i].wei;
            num++;
            tree[arr[i].from].push_back(i);
            tree[arr[i].to].push_back(i); 
        }
    }
    if(num==n-1)
        return rsl;
    else
        return 0;
}

exp=1/(n*(n-1))*(tot_len*2)

void dfs(int rt,int fa)
{
    sum[rt]=1;
    for(int i=0;i<tree[rt].size();i++){
        int& idx=tree[rt][i];        //treeæ•°ç»„ä¸­å­˜çš„æ˜¯è¾¹çš„ç¼–å·

            son=arr[idx].to;

            son=arr[idx].from;
        if(son==fa)        continue;
        dfs(son,rt);
        sum[rt]+=sum[son];
        ans+=(ll)((sum[son]*((ll)n-sum[son]))*(arr[idx].wei));
    }
}
