{
    int n;

    int a1,b1;
    cin>>a1>>b1;
    n--;

    if(!(a==0&&b==0)){
        a[0]=b[0]=0;
        a[1]=a1;b[1]=b1;
        pos=2;

        a[0]=b[0]=0;
        pos=1;
    }
    for(int i=0;i<n;i++,pos++)
        cin>>a[pos]>>b[pos];
    ll ans=0;
    int rightest=-1;
    for(int i=1;i<=n+1;i++){
        int lef1=a[i-1];
        int lef2=b[i-1];
        int rig1=a[i];
        int rig2=b[i];
        int lef=max(lef1,lef2);
        int rig=min(rig1,rig2);
        while(lef<=rightest)
            lef++;
        if(rig-lef>=0){
            rightest=rig;
            ans+=rig-lef+1;
        }
    }
    cout<<ans<<endl;
    return 0;
}
