int n,m;
const int maxn=205;
char nbo[maxn][maxn];
int kfc_dis1[maxn][maxn],kfc_dis2[maxn][maxn];
int yif[1][2],mer[1][2];

int vis[maxn][maxn];
typedef struct{
    int row;
    int col;
    int stp;
}pos;
bool check(pos xxx)
{
    if((nbo[xxx.row][xxx.col]=='M')||(nbo[xxx.row][xxx.col]=='Y')||(nbo[xxx.row][xxx.col]=='#')||
       (xxx.row>=n)||(xxx.row<0)||(xxx.col>=m)||(xxx.col<0)||(vis[xxx.row][xxx.col]==1))        
        return false;
    return true;
}
int kcnt=0;
void bfs(int stx,int sty)        //xä¸ºè¡Œ,yä¸ºåˆ— 
{    
    kcnt=0;
    queue<pos> que;
    pos now,next,sta;
    sta.row=stx;sta.col=sty;sta.stp=0;
    que.push(sta);
    vis[stx][sty]=1;
    memset(vis,0,sizeof(vis));
    while(!que.empty())
    {    
        now=que.front();
        if(nbo[now.row][now.col]=='@')    
            kfc_dis2[now.row][now.col]=now.stp;
        for(int i=0;i<4;i++)
        {    
            next.row=now.row+dir[i][0];
            next.col=now.col+dir[i][1];
            if(check(next))
            {    
                vis[next.row][next.col]=1;
                next.stp=now.stp+1;
                que.push(next);
            }
        }
        que.pop();
    }
    return ;
}
