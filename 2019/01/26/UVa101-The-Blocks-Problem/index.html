#include <iostream>

using namespace std;
vector<int> pile[30];
typedef pair<int, int> P;
int n;
P find_block(int num)
{
    P pos;
    for (int i=0; i < n; i++) {
        for (int j=0; j < pile[i].size(); j++) {
            if (pile[i][j] == num) {
                pos.first = i;
                pos.second = j;
            }
        }
    }
    return pos;
}

{

        int b = pile[p][i];
        pile[b].push_back(b);
    }
    pile[p].resize(h + 1);
}
