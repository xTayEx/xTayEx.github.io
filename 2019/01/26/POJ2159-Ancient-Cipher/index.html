#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <string.h>
using namespace std;
char ori[105];
char sec[105];
int na[30],nb[30];
int main()
{    
    int len1,len2;
    int i;
    int flag;
    while(scanf("%s %s",sec,ori)!=EOF)
    {    
        flag=1;
        len1=strlen(ori);
        len2=strlen(sec);
        for(i=0;i<len1;i++)
            na[ori[i]-'A']++;
        for(i=0;i<len2;i++)
            nb[sec[i]-'A']++;
        sort(na,na+26);sort(nb,nb+26);
        for(i=0;i<30;i++)
        {
            if(na[i]!=nb[i])
            {
                flag=0;
                break;
            }
        }
        if(flag==0)
        {
            printf("NO\n");
        }
        else
        {
            printf("YES\n");
        }
    }
    return 0;
}
