using namespace std;
const int maxn=1e5+5;
int head[maxn<<1];
int dep[maxn<<1];
int cur[maxn<<1];
int cnt;
struct edge{
    int to;int next;
    int wei;
};
edge es[maxn<<1];
inline void init()
{
    cnt=0;mst(head,-1);
    mst(dep,-1);mst(cur,-1);
}
void add(int u,int v,int wei)
{
    es[cnt].to=v;
    es[cnt].wei=wei;
    es[cnt].next=head[u];
    head[u]=cnt++;
}
void add_edge(int u,int v,int wei)
{
    add(u,v,wei);
    add(v,u,0);
}
bool bfs(int s,int t)
{
    queue<int> que;
    mst(dep,-1);dep[s]=0;
    que.push(s);
    while(!que.empty()){
        int u=que.front();
        que.pop();
        for(int i=head[u];i!=-1;i=es[i].next){
            if(es[i].wei>0&&dep[es[i].to]==-1){
                dep[es[i].to]=dep[u]+1;
                que.push(es[i].to);
            }
        }
    }
    return dep[t]!=-1;
}
int dfs(int s,int t,int mini)
{
    if(s==t) return mini;
    for(int &i=cur[s];i!=-1;i=es[i].next){
        if(es[i].wei>0&&dep[es[i].to]==dep[s]+1){
            int d=dfs(es[i].to,t,min(es[i].wei,mini));
            if(d>0){
                es[i].wei-=d;
                es[i^1].wei+=d;
                return d;
            }
        }
    }
    return 0;
}
int Dinic(int s,int t,int n)
{
    int ans=0;int tmp;
    while(bfs(s,t)){
        for(int i=0;i<=n;i++) cur[i]=head[i];
        while((tmp=dfs(s,t,INF))>0){
            ans+=tmp;
        }
    }
    return ans;
}

{
    int n,m,s,t;
    scanf("%d %d %d %d",&n,&m,&s,&t);
    init();
    for(int i=0;i<m;i++){
        int u,v,w;
        scanf("%d %d %d",&u,&v,&w);
        add_edge(u,v,w);
    }
    int ans=Dinic(s,t,n);

}
