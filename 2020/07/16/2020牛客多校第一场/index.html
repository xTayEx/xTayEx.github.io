inline void addedge(int u, int v, int cap, int cost)
{
    es[tol].to = v;
    es[tol].cap = cap;
    es[tol].cost = cost;
    es[tol].flow = 0;
    es[tol].nxt = head[u];
    head[u] = tol++;

    es[tol].to = u;
    es[tol].cap = 0;
    es[tol].cost = -cost;
    es[tol].flow = 0;

}
bool spfa(int s, int t)
{
    queue<int> que;
    for (int i = 0; i < N; i++) {
        dis[i] = INF;
        vis[i] = false;
        pre[i] = -1;
    }
    dis[s] = 0;
    vis[s] = true;
    que.push(s);
    while (!que.empty()) {
        int u = que.front();
        que.pop();
        vis[u] = false;
        for (int i = head[u]; i != -1; i = es[i].nxt) {
            int v = es[i].to;
            if (es[i].cap > es[i].flow && dis[v] > dis[u] + es[i].cost) {
                dis[v] = dis[u] + es[i].cost;
                pre[v] = i;
                if (!vis[v]) {
                    vis[v] = true;
                    que.push(v);
                }
            }
        }
    }
    if (pre[t] == -1)
        return false;
    else
        return true;
}
ll mcmf(int s, int t, ll& cost)
{
    ll flow = 0;
    cost = 0;
    while (spfa(s, t)) {
        ll Min = INF;
        for (int i = pre[t]; i != -1; i = pre[es[i ^ 1].to]) {
            if (Min > es[i].cap - es[i].flow) {
                Min = es[i].cap - es[i].flow;
            }
        }
        for (int i = pre[t]; i != -1; i = pre[es[i ^ 1].to]) {
            es[i].flow += Min;
            es[i ^ 1].flow -= Min;
            cost += es[i].cost * Min;
        }
        flow += Min;
        ansarr[(int)flow] = cost;
    }
    return flow;
}

ll exgcd(ll a, ll b, ll& x, ll& y)
{
    if (a == 0 && b == 0) {
        return -1;
    }
    if (b == 0) {
        x = 1;
        y = 0;
        return a;
    }
    ll d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}
ll inv(ll a, ll n)
{
    ll x, y;
    ll d = exgcd(a, n, x, y);
    if (d == 1)
        return (x % n + n) % n;
    else
        return -1;
}
